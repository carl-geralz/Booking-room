Table ActivityLog {
    activityLogId string
    activityLogTimestamp datetime
    employee_id string [ref: > Employee.employeeId]
    activityLogDescription string
}

Table Employee {
    employeeId string
    employeeName string
    department string
    phoneNumber string
    corporateEmail string
    user_id string [ref: > User.userId]
}

Table Equipment {
    equipmentId string
    equipmentName string
    reservation_id string [ref: > Reservation.reservationId]
}

Table Reservation {
    reservationId string
    employee_id string [ref: > Employee.employeeId]
    room_id string [ref: > Room.roomId]
    reserveDate datetime
    startTime datetime
    endTime datetime
    reservationStatus reservationStatus [note: "Enum to reservation status"]
    descriptionByUser string
    descriptionByAdmin string
}

Table Role {
    roleId string
    constantRole string
}

Table Room {
    roomId string
    roomType string
    roomCapacity string
}

Table RoomFacility {
    roomFacilityId string
    roomFacilityName string
    roomId string [ref: <> Room.roomId]
}

Table User {
    userId string
    username string
    password string
    role_id string [ref: > Role.roleId]
}

Enum reservationStatus {
  "PENDING"
  "APPROVED"
  "DECLINED"
  "CANCELLED"
}
